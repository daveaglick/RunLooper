00122         // Processes one event in the queue and returns false if the queue was cancelled
00123         private bool Process(bool block)
00124         {
00125             // If we've requested to cancel the run loop, just return
00126             if (_cancel.IsCancellationRequested) return false;
00127             
00128             // Try to get an item off the queues in priority order without blocking
00129             IRunLoopItem item = null;
00130             if (_queues.FirstOrDefault(c => c.TryTake(out item)) == null)
00131             {
00132                 // If we don't want to block, just return
00133                 if (!block) return false;
00134 
00135                 // No items to take, so block until one of the queues gets an item
00136                 try
00137                 {
00138                     BlockingCollection<IRunLoopItem>.TakeFromAny(_queues, out item, _cancel.Token);
00139                 }
00140                 catch (OperationCanceledException)
00141                 {
00142                     // If we've canceled, return false
00143                     return false;
00144                 }
00145             }
00146 
00147             // Execute the item
00148             item.Execute();
00149             return true;
00150         }

00253         internal void Enqueue(IRunLoopItem item, Priority priority)
00254         {
00255             if (_cancel.IsCancellationRequested) throw new ObjectDisposedException("RunLoop");
00256             if (item == null) throw new ArgumentNullException("item");
00257 
00258             // If we're on the main thread and this is a synchronous request, execute immediately
00259             // otherwise we'll get a deadlock while the caller waits for the item to process in
00260             // the queue but never returns control to the RunLoop to actually process the queue
00261             if(item.Synchronous && Thread.CurrentThread.ManagedThreadId == ManagedThreadId)
00262             {
00263                 item.Execute();
00264                 return;
00265             }
00266 
00267             // We're not on the main thread, queue for execution
00268             _queues[(int)priority].Add(item);
00269         }
